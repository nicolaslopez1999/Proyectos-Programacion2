package javaapplication25;

import javax.swing.JOptionPane;
/**
 *Tabla de multiplicar (While)
 * @Nicolas Lopez
 */
public class JavaApplication25 {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
        
        int x=1,i,y=1;
        while (y<=10){
            JOptionPane.showMessageDialog(null,"TABLA #"+y);
            i=1;  //"i" se reinicia en 1 para que el ciclo empieze de nuevo
            while (i<=10){
            JOptionPane.showMessageDialog(null,x+" x "+i+" = "+x*i);
            i++;
        }
            x++;
            y++;
        }
        System.exit(0);
    }
    
}






package tabla;

import javax.swing.JOptionPane;
/**
 *Tablas de multiplicar (ciclo For)
 * @author Nicolas Lopez
 */
public class Tabla {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
        int x,y,total;
        for (x=1;x<=10;x++){
            JOptionPane.showMessageDialog(null,"TABLA No."+x);
            for (y=1;y<=10;y++){
                total=x*y;
                JOptionPane.showMessageDialog(null,x+"X"+y+" = "+total);
            }
        }
        System.exit(0);
        
    }
}
